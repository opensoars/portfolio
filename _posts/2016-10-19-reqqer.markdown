## Install

`npm install reqqer`


## Use

ES 2015
{% highlight javascript %}
// Using an object
var reqs = require('reqqer')({
  https: 'https',
  http: 'http'
});

// Using an array
var reqs = require('reqqer')(['https', 'http']);

var https = reqs.https,
    http = reqs.http;
{% endhighlight %}


ES 2016
{% highlight javascript %}
// Using an object
const {https, http} = require('reqqer')({
  https: 'https',
  http: 'http'
});

// Using an array
const {https, http} = require('reqqer')(['https', 'http']);
{% endhighlight %}
